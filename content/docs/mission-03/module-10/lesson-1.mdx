**Basics** of reading Files with the fs Module

Check if node is working

```js title='hello.js'
console.log("node version :", process.version);
console.log("platform :", process.platform);
```

args in node.js

```js title="greetings.js"
const args = process.argv;

// process.arg[0] = node path
// process.arg[1] = file path
// process.arg[2] = first argument

const userName = args[2] || "guest";
const time = new Date().getHours();

let greeting;

if (time < 12) {
  greeting = "Good Morning";
} else if (time < 18) {
  greeting = "Good Afternoon";
} else {
  greeting = "Good Evening";
}

console.log(`${greeting} ${userName}`);
```

Reading files synchronously

```js title="read-sync.js"
const fs = require("fs");

console.log("Start Reading...");

try {
  const data = fs.readFileSync("./data/diary.txt", "utf-8");
  console.log("file content");
  console.log(data);
} catch (err) {
  console.error(err.message);
}

console.log("Finished");
```

Reading files asynchronously

```js title="read-async.js"
const fs = require("fs");

console.log("Start Reading...");

fs.readFile("./data/diary.txt", "utf-8", (error, data) => {
  if (error) {
    console.error(error.message);
  } else {
    console.log("file content: ");
    console.log(data);
  }
});

console.log("This runs emmideatly - without blocking");
```
