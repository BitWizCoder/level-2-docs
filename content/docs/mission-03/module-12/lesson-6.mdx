GET all users 

```typescript title="./src/server.js"
app.get("/users", async (req: Request, res: Response) => {
  try {
    const result = await pool.query(`
      SELECT * FROM users
      `);

    res.status(200).json({
      success: true,
      message: "Users retrived successfully!!",
      data: result.rows,
    });
  } catch (error: any) {
    res.status(500).json({
      success: false,
      message: error.message,
      details: error,
    });
  }
});
```

GET single user using params

```typescript title="./src/server.js"
app.get("/users/:id", async (req: Request, res: Response) => {
  console.log(req.params.id);

  try {
    const result = await pool.query(`SELECT * FROM users WHERE id = $1`, [
      req.params.id,
    ]);

    if (result.rows.length === 0) {
      res.status(404).json({
        success: false,
        message: "User not found!!",
      });
    } else {
      res.status(200).json({
        success: true,
        message: "User retrived successfully!!",
        data: result.rows[0],
      });
    }
  } catch (error: any) {
    res.status(500).json({
      success: false,
      message: error.message,
      details: error,
    });
  }
});
```
